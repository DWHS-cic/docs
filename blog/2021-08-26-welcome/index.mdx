#include <iostream>
using namespace std;

int main(){
    int n, m, k, all = 0;
    cin >> n >> m;
    int age[n];
    for (int i = 0; i < n; i++) {
        cin >> age[i];
        all += m;
    }
    int li, ri, pi;
    cin >> k;
    for (int i = 0; i < k; i++) {
        cin >> li >> ri >> pi;
        for (int j = 0; j < n; j++) {
            if (li <= age[j] && age[j] <= ri) {
                all -= (m - pi);
            }
        }
    }
    cout << all << endl;
    return 0;
}
